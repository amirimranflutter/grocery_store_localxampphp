SELECT o.order_id, c.customer_name, e.emp_name as cashier, p.p_name as product, cat.cat_name as category, s.supplier_name, oi.quantity, oi.subtotal FROM orders o JOIN customers c ON o.customer_id = c.customer_id JOIN employees e ON o.emp_id = e.emp_id JOIN order_items oi ON o.order_id = oi.order_id JOIN products p ON oi.p_id = p.p_id JOIN categories cat ON p.cat_id = cat.cat_id JOIN suppliers s ON p.supplier_id = s.supplier_id; -- This query joins 7 tables!
